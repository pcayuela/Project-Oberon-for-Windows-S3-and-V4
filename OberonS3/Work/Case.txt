MODULE Case;IMPORT In,Out;VAR countVowels: INTEGER; ch: CHAR;PROCEDURE Vowcount *;BEGIN  In.Open;  countVowels:=0;  LOOP    In.Char(ch);    IF ~In.Done THEN EXIT END;    CASE ch OF      "a", "e", "i", "o", "u",      "A", "E","I","O","U": INC(countVowels)    ELSE    END;  END;  Out.Int(countVowels,0);  Out.String(" vowels read.");Out.Ln;	END Vowcount;END Case.Case.Vowcount
